#include <stdio.h>

void triangle(int n) {
    int spaces = 2 * (n - 1);
    
    for (int i = 1; i <= n; i++) {
       
        printf("%d", i);
        for (int j = 0; j < i; j++) {
            printf("*");
        }
        printf("%d", i);

        for (int j = 0; j < spaces; j++) {
            printf(" ");
        }

        int star_count = 2 * i - 1; 
        int number = 2 * i - 1; 
        printf("%d", number);
        for (int j = 0; j < star_count; j++) {
            printf("*");
        }
        printf("%d", number);

       
        for (int j = 0; j < spaces; j++) {
            printf(" ");
        }
        printf("%d", i);

        for (int j = 0; j < i; j++) {
            printf("*");
        }
        printf("%d", i);

       
        printf("\n");

        spaces -= 2; 
    }

   
    spaces = 2;

    for (int i = n - 1; i >= 1; i--) {
       
        printf("%d", i);
        for (int j = 0; j < i; j++) {
            printf("*");
        }
        printf("%d", i);

 
        for (int j = 0; j < spaces; j++) {
            printf(" ");
        }

        int star_count = 2 * i - 1; 
        int number = 2 * i - 1; 
        printf("%d", number);
        for (int j = 0; j < star_count; j++) {
            printf("*");
        }
        printf("%d", number);

   
        for (int j = 0; j < spaces; j++) {
            printf(" ");
        }
        printf("%d", i);

        for (int j = 0; j < i; j++) {
            printf("*");
        }
        printf("%d", i);

        
        printf("\n");

        spaces += 2; 
    }
}
